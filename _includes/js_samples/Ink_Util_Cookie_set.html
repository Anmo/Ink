Set a cookie that expires in one day

Code

{% highlight js %} Ink.requireModules(['Ink.Util.Cookie_1'], function( InkCookie ){ InkCookie.set( 'inkcookie', 'one_day_ttl', (60 * 60 * 24)); }); {% endhighlight %}
