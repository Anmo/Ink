Code

{% highlight javascript %} Ink.requireModules(['Ink.UI.Animate_1', 'Ink.Dom.Element_1'], function (Animate, InkElement) { new Animate('#some-element', { animation: 'fadeOut', trigger: '#some-button' onEnd: function (endData) { InkElement.remove(endData.element); } }); }); {% endhighlight %}
